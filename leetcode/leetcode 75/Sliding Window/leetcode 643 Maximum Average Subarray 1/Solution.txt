class Solution:
    def findMaxAverage(self, nums: List[int], k: int) -> float:
        sums=0
        for i in range(k):
            sums+=nums[i]
        currSum = maxSum = sums

        for j in range(k,len(nums)):
            currSum+=nums[j]-nums[j-k]
            if currSum>maxSum:
                maxSum = currSum
        
        return maxSum/k
        
	-OR-

	currSum = maxSum = sum(nums[:k])


        for i in range(k, len(nums)):
            currSum+=nums[i]-nums[i-k]

            maxSum = max(maxSum, currSum)
            

            
        return  maxSum /k
