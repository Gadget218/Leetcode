class Solution:
    def longestOnes(self, nums: List[int], k: int) -> int:
        l=0
        count=0
        ln=0
        max_ln=0
        for r in range(len(nums)):
            if nums[r]==0:
                count+=1
            while count>k:
                if nums[l]==0:
                    count-=1
                ln-=1
                l+=1
            ln+=1
            max_ln=max(max_ln,ln)
        return max_ln
