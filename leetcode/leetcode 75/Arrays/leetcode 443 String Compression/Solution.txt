class Solution:
    def compress(self, chars: List[str]) -> int:
        write = 0
        i = 0

        while i<len(chars):
            char = chars[i]
            count = 0

            #count the group
            while i<len(chars) and chars[i]==char:
                i+=1
                count+=1

            #write character
            chars[write] = char
            write +=1

            #write the count if
            if count>1:
                for n in str(count):
                    chars[write]=n
                    write+=1
        return write  

T: O(n)
S: O(1)
