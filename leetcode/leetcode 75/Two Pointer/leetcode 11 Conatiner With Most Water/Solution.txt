class Solution:
    def maxArea(self, height: List[int]) -> int:
        max_vol = 0
        i=0
        j=len(height)-1
        while i<j:
            vol = min(height[j],height[i])*(j-i)
            if vol>max_vol:
                max_vol = vol
            if height[i]<height[j]:
                i+=1
            else:
                j-=1
        return max_vol
        

T:O(n)
S:O(1)
